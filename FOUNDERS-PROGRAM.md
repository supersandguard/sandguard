#### Smart Contract Outline

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract SandGuardFounder is ERC721, Ownable {
    uint256 public constant MAX_SUPPLY = 100;
    uint256 public totalMinted;
    
    mapping(uint256 => uint256) public joinTimestamp;
    mapping(uint256 => string) public founderName; // optional ENS/username
    
    // ERC-5192: Soulbound
    event Locked(uint256 tokenId);
    
    constructor() ERC721("SandGuard Founder", "SGFOUNDER") Ownable(msg.sender) {}
    
    function mint(address to, uint256 founderNumber) external onlyOwner {
        require(founderNumber >= 1 && founderNumber <= 100, "Invalid number");
        require(totalMinted < MAX_SUPPLY, "All 100 minted");
        
        _mint(to, founderNumber);
        joinTimestamp[founderNumber] = block.timestamp;
        totalMinted++;
        
        emit Locked(founderNumber); // Soulbound signal
    }
    
    // Override transfers to make soulbound
    function _update(address to, uint256 tokenId, address auth) 
        internal override returns (address) 
    {
        address from = _ownerOf(tokenId);
        // Allow minting (from == address(0)) but block transfers
        require(from == address(0), "SandGuard Founder tokens are soulbound");
        return super._update(to, tokenId, auth);
    }
    
    // ERC-5192: locked() returns true for all tokens
    function locked(uint256 tokenId) external view returns (bool) {
        require(_ownerOf(tokenId) != address(0), "Token does not exist");
        return true;
    }
    
    // Owner can set founder display name
    function setFounderName(uint256 tokenId, string calldata name) external {
        require(ownerOf(tokenId) == msg.sender, "Not your token");
        founderName[tokenId] = name;
    }
    
    // Recovery: owner can force-transfer for compromised wallets
    function recover(uint256 tokenId, address newOwner) external onlyOwner {
        // Bypasses soulbound restriction for recovery only
        _transfer(_ownerOf(tokenId), newOwner, tokenId);
    }
    
    function tokenURI(uint256 tokenId) public view override returns (string memory) {
        require(_ownerOf(tokenId) != address(0), "Token does not exist");
        // Return metadata URI (IPFS or on-chain SVG)
        return string(abi.encodePacked(
            "https://supersandguard.com/api/founders/metadata/",
            Strings.toString(tokenId)
        ));
    }
}
```

### Is This Sustainable?

**Yes, and here's why:**

1. **Forgone revenue is theoretical.** At launch, SandGuard has ~0 paying users. Giving away 100 Pro subscriptions to people who wouldn't have paid anyway costs nothing real.

2. **100 active users is worth more than 100 Ã— $20/mo.** The social proof, feedback, testimonials, and referrals from 100 committed Founders will generate far more than $2,000/month in downstream value.

3. **$UMBRA allocation is small.** 5M tokens out of 1B (0.5% of supply) allocated to the 100 most committed users is extremely reasonable.

4. **The program self-limits.** 100 Founders, then it closes. No ongoing dilution. No escalating costs.

5. **Founders become the referral engine.** With 2x referral rewards, Founders will bring paid users who more than compensate for the forgone revenue.

### Break-Even Analysis

If each Founder refers just **1 paid user** who stays for 6 months:
- Revenue from referred users: 100 Ã— $20/mo Ã— 6 months = **$12,000**
- "Cost" of Founder program: 100 Ã— $240/yr = $24,000 forgone (theoretical)
- Net: Already recovered 50% of theoretical cost from referrals alone

In reality, many Founders will refer multiple users, and the social proof / governance / community benefits have value far beyond direct referrals.
